First Step: One time setup
git config --global user.name "Your Name"
git config --global user.email "your@email.com"
git config --list

Second Step:Start a Repo
git init                  
git clone <url>     

Third Step:Working with Files
git status              
git add <file>           
git add .                
git commit -m "message"  

Fourth Step: Branching
git branch              
git branch <name>        
git checkout <name> 
git checkout -b <name>  
git merge <branch>


Fifth Step: Connect to Remote
git remote add origin<url>
git remote -v

Sixth Step:Push and Pull
git push -u origin main  #for first push
git push                 # Push changes
git pull                 # Get latest changes from remote

Sevent Step: Undo/Fix
git restore <file>            # Discard file changes
git reset <file>              # Unstage file
git reset --hard HEAD         # Reset all changes 
git revert <commit_id>        # Undo commit safely
